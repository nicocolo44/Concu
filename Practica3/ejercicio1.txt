process lectores[id:1..N]{
    baseDeDatos.permiso();
    //Leo base de datos
    baseDeDatos.termine();
}

monitor baseDeDatos{                    //Creo que es mejor nombrarlo como adminBaseDeDatos
    int cantLectores = 0;               
    cond espera;
    int esperando = 0;

    Procedure permiso(){
        if(cantLectores == 5){
            esperando++;
            wait(espera);
        }
        cantLectores++;
    }

    Procedure termine(){
        if(esperando == 0){
            cantLectores--;
        }else{
            esperando--;
            signal(espera);
        }
    }
}
